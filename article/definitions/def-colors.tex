\ExplSyntaxOn
\NewDocumentCommand{\disableStrategy}{ m }{%
  \str_case:nn {#1}
    {
      {1}{\ShowStrategyOnefalse}
      {2}{\ShowStrategyTwofalse}
      {3}{\ShowStrategyThreefalse}
      {4}{\ShowStrategyFourfalse}
      {5}{\ShowStrategyFivefalse}
    }%
}
\NewDocumentCommand{\enableStrategy}{ m }{%
  \str_case:nn {#1}
    {
      {1}{\ShowStrategyOnetrue}
      {2}{\ShowStrategyTwotrue}
      {3}{\ShowStrategyThreetrue}
      {4}{\ShowStrategyFourtrue}
      {5}{\ShowStrategyFivetrue}
    }%
}
\NewDocumentCommand{\StrategyLabel}{ m m m O{long} }{%
  {\setlength{\fboxsep}{0.45pt}\fcolorbox{#2}{#2!22}{\scriptsize\strut #1}}%
  \str_if_eq:nnF {#4} {short} {\nobreakspace{}#3}%
}
\NewDocumentCommand{\strategy}{ m O{long} }{%
  \str_case:nnF {#1}
    {
      {1}{\StrategyLabel{1}{CentralMassiveMimoColor}{Central~massive~MIMO}[#2]}
      {2}{\StrategyLabel{2}{DistributedFixedColor}{Distributed~fixed}[#2]}
      {3}{\StrategyLabel{3}{DistributedMovableColor}{Distributed~movable~(Opt.~1)}[#2]}
      {4}{\StrategyLabel{4}{DistributedMovableOptTwoColor}{Distributed~movable~(Opt.~2)}[#2]}
      {5}{\StrategyLabel{5}{DistributedMovableOptThreeColor}{Distributed~movable~(Opt.~3)}[#2]}
    }
    {(#1)}%
}
\ExplSyntaxOff
